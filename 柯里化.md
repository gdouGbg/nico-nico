      // 实现函数
      var foo = function (...args) {
        // 创建数组存储参数
        let _args = [...args];
        return foo;
      };
      foo.prototype.getValue = () => {
        return foo._args.reduce((acc, cur) => {
          return acc + cur;
        }, 0);
      };

      var f1 = foo(1, 2, 3);
      f1.getValue(); // 6
      var f2 = foo(1)(2, 3);
      f2.getValue(); // 6
      var f3 = foo(1)(2)(3)(4);
      f3.getValue(); // 10
